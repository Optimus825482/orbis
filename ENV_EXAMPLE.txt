# Environment Variables Example
# BU DOSYAYI .env OLARAK KOPYALAYIN VE DEĞERLERİ DOLDURUN

# ===========================================
# OBLIGATORY (ZORUNLU) DEĞİŞKENLER
# ===========================================

# OpenCage Geocoding API
# Ücretsiz API key almak için: https://opencagedata.com/api
OPENCAGE_API_KEY=your-opencage-api-key-here

# AI Provider API Keys - En az biri gereklidir
# Hyperbolic API (https://hyperbolic.xyz/)
HYPERBOLIC_API_KEY=your-hyperbolic-api-key-here

# Google Gemini API (https://ai.google.dev/)
GEMINI_API_KEY=your-gemini-api-key-here

# ===========================================
# OPTIONAL (OPSİYONEL) DEĞİŞKENLER
# ===========================================

# Database Configuration
DATABASE_URL=sqlite:///instance/astro.db
# PostgreSQL için: postgresql://user:password@localhost/dbname

# Session Management
SESSION_TYPE=filesystem  # Gelecekte: redis
SESSION_COOKIE_SECURE=False  # Production için: True
SESSION_LIFETIME_HOURS=1

# Supabase (Opsiyonel - Eğer kullanıyorsanız)
SUPABASE_URL=your-supabase-project-url
SUPABASE_KEY=your-supabase-anon-key
SUPABASE_SERVICE_KEY=your-supabase-service-key

# OpenRouter API (Opsiyonel - Backup AI provider)
OPENROUTER_API_KEY=your-openrouter-api-key

# Flask Configuration
FLASK_DEBUG=False
FLASK_TESTING=False
FLASK_ENV=production

# ===========================================
# GELİŞTİRME NOTLARI
# ===========================================
# 
# 1. Bu dosyayı .env olarak kopyalayın
# 2. Tüm "your-*" değerlerini gerçek API anahtarlarınızla değiştirin
# 3. .env dosyasını asla GIT'e commit etmeyin (.gitignore'da)
# 4. Production için SESSION_COOKIE_SECURE=True yapın
